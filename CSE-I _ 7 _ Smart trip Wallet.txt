  const pad = (n) => String(n).padStart(2, '0');

  const hh = pad(d.getHours());   // ← added ()
  const mm = pad(d.getMinutes()); // ← added ()

  document.getElementById('clock').textContent = hh + ':' + mm;
}

function deriveTierFromBudget(totalBudget, days) {
  totalBudget = Number(totalBudget) || 0;
  days = Number(days) || 0;

  if (totalBudget <= 0 || days <= 0) return null;

  const perDay = totalBudget / days;

  if (perDay < 2000) return 'budget';
  if (perDay < 4000) return 'mid';
  return 'comfort';
}


function currentHotelTier(){

if (!currentTrip) return 'mid';

const days = dayDiff(currentTrip.start, currentTrip.end);

const byBudget = deriveTierFromBudget(Number(currentTrip.budget||0), days);

return currentTrip.tier || byBudget || 'mid';

}

async function sha256(text){

if (window.crypto?.subtle) {

const enc = new TextEncoder().encode(text);

const buf = await crypto.subtle.digest('SHA-256', enc);

return Array.from(new Uint8Array(buf)).map(b=>b.toString(16).padStart(2,'0')).join('');

} else {

return btoa(text);

}

}

function dayDiff(start,end){ return Math.max(1, Math.round((new Date(end) - new Date(start))/(1000*60*60*24))+1 );}

function buildPlan(cityKey, days){

const city = DATA[cityKey] || null;

if (!city){

const genericSights = [

{ name:"Old Town walking tour", slot:"morning", time:"2–3h"},

{ name:"City Museum / Gallery", slot:"afternoon", time:"2h"},

{ name:"Popular market & local street food", slot:"evening", time:"2h"},

{ name:"Central park or viewpoint", slot:"morning", time:"2h"},

{ name:"Day trip to nearby attraction", slot:"full", time:"6–8h"},

];

return {

city:null,

daily:Array.from({length:days},(_,i)=>{

const d=i+1;

if (days>=3 && d===Math.ceil(days/2)){

return { day:d, morning:genericSights[4], afternoon:{name:"Local café / brunch", time:"Flexible"}, evening:{name:"Night market", time:"2h"} };

}

return { day:d, morning:genericSights[0], afternoon:genericSights[1], evening:genericSights[2] };

})

};

}

const pool = [...city.sights];

const pick = (slot)=>{

const idx = pool.findIndex(x=>x.slot===slot);

if (idx>-1) return pool.splice(idx,1)[0];

return pool.length ? pool.splice(0,1)[0] : { name:"Free time", slot, time:"Flexible" };

};

const daysArr = [];

for (let d=1; d<=days; d++){

if (pool.find(p=>p.slot==='full') && days>=3 && d===Math.ceil(days/2)){

const fullIndex = pool.findIndex(p=>p.slot==='full');

const fullItem = pool.splice(fullIndex,1)[0];

daysArr.push({ day:d, morning:fullItem, afternoon:{name:"Leisure / Café break", time:"Flexible"}, evening:{name:"Easy evening stroll / shopping", time:"1–2h"} });

} else {

daysArr.push({ day:d, morning:pick('morning'), afternoon:pick('afternoon'), evening:pick('evening') });

}

}

return { city, daily:daysArr };

}

async function geocodePlace(query) {
try {
const url = `https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(query)}`;

const res = await fetch(url, {
headers: {
'Accept-Language': 'en'
}
});

const data = await res.json();

if (data && data.length > 0) {
return {
lat: parseFloat(data[0].lat),
lng: parseFloat(data[0].lon)
};
}

return null;
} catch (err) {
console.error("Geocoding error:", err);
return null;
}
}


async function plotTripOnMap() {

if (!map || !currentTrip) return;


mapMarkers.forEach(m => map.removeLayer(m));

mapMarkers = [];


const cityKey = norm(currentTrip.place);

const city = DATA[cityKey];

if (!city) return;


const bounds = [];


(city.sights || []).forEach(p => {

if (!p.lat || !p.lng) return;


const marker = L.marker([p.lat, p.lng]).addTo(map);

marker.bindPopup(`

<div style="width:160px">

<img src="${p.img}"

style="width:100%;height:90px;object-fit:cover;border-radius:6px">

<b>${p.name}</b><br>

<small>${p.area || ""}</small>

</div>

`);


mapMarkers.push(marker);

bounds.push([p.lat, p.lng]);

});


for (const h of (city.hotels || [])) {

let lat = h.lat;
let lng = h.lng;

if (typeof lat !== 'number' || typeof lng !== 'number') {
const result = await geocodePlace(h.name + " " + (cityKey || ""));
if (result) {
lat = result.lat;
lng = result.lng;
} else {
continue;
}
}

const marker = L.marker([lat, lng]).addTo(map);

marker.bindPopup(`
<div style="width:160px">
<img src="${h.img || 'https://via.placeholder.com/160x90'}"
style="width:100%;height:90px;object-fit:cover;border-radius:6px">
<b>🏨 ${h.name}</b><br>
<small>${h.area || ""}</small>
</div>
`);

mapMarkers.push(marker);
bounds.push([lat, lng]);
}


if (bounds.length) {

map.fitBounds(bounds, { padding: [40, 40] });

}

}
